CREATE TABLE Candidate (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255),
    age INT,
    DateOfBirth DATE,
    qualification VARCHAR(255),
    assets_value DECIMAL(15, 2),
    party VARCHAR(255),
    address VARCHAR(255),
    district VARCHAR(255),
    state VARCHAR(255),
    country VARCHAR(255)
);

CREATE TABLE Election (
    id INT AUTO_INCREMENT PRIMARY KEY,
    date_of_announcement DATE,
    year INT,
    results_date DATE,
    state VARCHAR(255),
    country VARCHAR(255)
);

CREATE TABLE Results (
    id INT AUTO_INCREMENT PRIMARY KEY,
    constitution_name VARCHAR(255),
    candidate_name VARCHAR(255),
    party_name VARCHAR(255),
    results_date DATE,
    votes_acquired INT,
    position_acquired_in_constitution INT,
    deposit_hold ENUM('yes', 'no'),
    district VARCHAR(255),
    state VARCHAR(255),
    country VARCHAR(255)
);
